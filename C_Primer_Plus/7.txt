#include<stdio.h>
#include<ctype.h>
#include<stdbool.h>
#define STOP '|'
int main(void)
{
	char c;
	char prev;
	long n_chars = 0L;
	int n_lines = 0, n_words = 0, p_lines = 0;
	bool inword = false;
	printf("Enter text to be analyzed(| to terminate):\n");
	prev = '\n';
	while ((c = getchar()) != STOP)
	{
		n_chars++;
		if (c == '\n')
			n_lines++;
		if (!isspace(c) && !inword)
		{
			inword = true;
			n_words++;
		}
		if (isspace(c) && inword)
			inword = false;
		prev = c;

	}
	if (prev != '\n')
		p_lines = 1;
	printf("Characters =%ld,words = %d,lines = %d", n_chars, n_words, n_lines);
	printf("partial lines = %d\n", p_lines);
	return 0;
}

{
	double hours, rate, get, gross;
	printf("Please input your work hours per week:\n");
	scanf("%lf", &hours);
	if (hours > TIME)
		hours = TIME + (hours - TIME)*OT;
	gross = hours*BASIC;
	if (gross <= LIMIT1)
		rate = RATE1*gross;
	else if (gross <= (LIMIT1 + LIMIT2))
		rate = RATE11 + (gross - LIMIT1)*RATE2;
	else

    get = gross - rate;
	printf("Your salary in total is %.2f,your rate is %.2f,your net income is %.2f\n", gross, rate, get);
